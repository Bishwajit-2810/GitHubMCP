07 Constraints & Foreign Keys 
CREATE TABLE orders (
  id   INT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
  uid  INT NOT NULL
       REFERENCES users(id)
       ON DELETE CASCADE
       ON UPDATE CASCADE,
  amount  NUMERIC(10,2) CHECK (amount > 0),
  status  TEXT CHECK (
    status IN ('pending','paid','cancelled')
  ) DEFAULT 'pending',
  UNIQUE (uid, amount)  -- composite unique
);

08 Joins
Join Types
-- INNER JOIN (matching rows both sides)
SELECT u.name, o.amount
FROM users u
JOIN orders o ON u.id = o.uid;

-- LEFT JOIN (all users, matched orders)
SELECT u.name, o.amount
FROM users u
LEFT JOIN orders o ON u.id = o.uid;

-- RIGHT JOIN
SELECT u.name, o.amount
FROM users u
RIGHT JOIN orders o ON u.id = o.uid;

-- FULL OUTER JOIN
SELECT u.name, o.amount
FROM users u
FULL JOIN orders o ON u.id = o.uid;

-- CROSS JOIN (Cartesian)
SELECT * FROM colors CROSS JOIN sizes;
Multi-Table & Self Join
-- Three-table join
SELECT u.name, o.amount, p.name AS product
FROM users u
JOIN orders o ON u.id = o.uid
JOIN products p ON o.product_id = p.id;

-- Self join (e.g., manager)
SELECT e.name, m.name AS manager
FROM employees e
LEFT JOIN employees m
  ON e.manager_id = m.id;

-- LATERAL join
SELECT u.name, recent.*
FROM users u,
LATERAL (
  SELECT * FROM orders
  WHERE uid = u.id
  ORDER BY created_at DESC
  LIMIT 1
) recent;

14 Functions & Stored Procedures
PL/pgSQL Function
CREATE OR REPLACE FUNCTION
  get_user_total(p_uid INT)
RETURNS NUMERIC AS $$
DECLARE
  total NUMERIC;
BEGIN
  SELECT COALESCE(SUM(amount), 0)
    INTO total
  FROM orders
  WHERE uid = p_uid;

  RETURN total;
END;
$$ LANGUAGE plpgsql;

-- Call it
SELECT get_user_total(1);
Returns Table
CREATE FUNCTION active_orders(lim INT)
RETURNS TABLE(
  order_id INT,
  user_name TEXT,
  amount NUMERIC
) AS $$
BEGIN
  RETURN QUERY
    SELECT o.id, u.name, o.amount
    FROM orders o
    JOIN users u ON o.uid = u.id
    LIMIT lim;
END;
$$ LANGUAGE plpgsql;

-- Stored Procedure (no return)
CREATE PROCEDURE archive_orders()
LANGUAGE plpgsql AS $$
BEGIN
  INSERT INTO archive
    SELECT * FROM orders
    WHERE status = 'paid';
  DELETE FROM orders
    WHERE status = 'paid';
END;
$$;
CALL archive_orders();
15 Triggers 
-- Trigger function
CREATE OR REPLACE FUNCTION
  update_timestamp()
RETURNS TRIGGER AS $$
BEGIN
  NEW.updated_at = NOW();
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- Attach trigger
CREATE TRIGGER trg_users_updated
BEFORE UPDATE ON users
FOR EACH ROW
EXECUTE FUNCTION update_timestamp();
-- Audit log trigger
CREATE FUNCTION log_delete()
RETURNS TRIGGER AS $$
BEGIN
  INSERT INTO audit_log
    (tbl, action, old_data, at)
  VALUES
    (TG_TABLE_NAME, 'DELETE',
     row_to_json(OLD), NOW());
  RETURN OLD;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_audit
AFTER DELETE ON users
FOR EACH ROW
EXECUTE FUNCTION log_delete();

19 CTE, Window Functions & Subqueries
Common Table Expressions
-- Simple CTE
WITH high_earners AS (
  SELECT * FROM employees
  WHERE salary > 80000
)
SELECT * FROM high_earners;

-- Chained CTEs
WITH
  base AS (
    SELECT uid, SUM(amount) total
    FROM orders GROUP BY uid
  ),
  ranked AS (
    SELECT *,
      RANK() OVER
        (ORDER BY total DESC)
    FROM base
  )
SELECT * FROM ranked WHERE rank <= 10;

-- Recursive CTE (org chart)
WITH RECURSIVE org AS (
  SELECT id, name, manager_id
  FROM employees
  WHERE manager_id IS NULL
  UNION ALL
  SELECT e.id, e.name, e.manager_id
  FROM employees e
  JOIN org o ON e.manager_id = o.id
)
SELECT * FROM org;
Window Functions
SELECT
  name, dept, salary,

  -- Rank within department
  RANK() OVER (
    PARTITION BY dept
    ORDER BY salary DESC
  ) AS dept_rank,

  -- Row number (unique)
  ROW_NUMBER() OVER
    (ORDER BY salary DESC) AS row_n,

  -- Running total
  SUM(salary) OVER (
    PARTITION BY dept
    ORDER BY salary
  ) AS running_total,

  -- Percent rank
  PERCENT_RANK() OVER
    (ORDER BY salary) AS pct,

  -- Lead / Lag
  LAG(salary, 1) OVER
    (ORDER BY id) AS prev_salary,
  LEAD(salary, 1) OVER
    (ORDER BY id) AS next_salary

FROM employees;

V3 Distance Operators
Operator Distance Metric Best Used For Example
<-> Euclidean (L2) General similarity, spatial embedding <-> query_vec
<=> Cosine distance NLP embeddings (direction matters) embedding <=> query_vec
<#> Inner product (neg) Normalized vectors, dot product embedding <#> query_vec
<+> L1 (Manhattan) Sparse vectors (v0.7.0+) embedding <+> query_vec
Compute Distances
-- L2 distance
SELECT id,
  embedding <-> '[1,2,3]'::vector
    AS l2_dist
FROM items
ORDER BY l2_dist LIMIT 5;

-- Cosine similarity = 1 - cosine_distance
SELECT id,
  1 - (embedding <=>
    '[1,2,3]'::vector)
  AS cosine_similarity
FROM items
ORDER BY cosine_similarity DESC;

V4 Vector Indexes — IVFFlat & HNSW
IVFFlat
Inverted File with Flat compression. Divides vectors into lists (Voronoi cells). Approximate nearest neighbor via list probing. 
-- Build after inserting data
CREATE INDEX ON documents
USING ivfflat (embedding vector_cosine_ops)
WITH (lists = 100);

-- Set probes at query time
SET ivfflat.probes = 10;

-- Rule of thumb:
-- lists = rows / 1000 (up to 1M)
-- lists = sqrt(rows) (over 1M)
-- probes: higher = slower + accurate
HNSW (recommended)
Hierarchical Navigable Small World graphs. Better recall at same speed. Slower to build, uses more memory. Preferred for production. 
-- Create HNSW index
CREATE INDEX ON documents
USING hnsw (embedding vector_cosine_ops)
WITH (m = 16, ef_construction = 64);

-- Tune at query time
SET hnsw.ef_search = 100;

-- m: neighbors per layer (16 default)
-- ef_construction: build accuracy (64)
-- ef_search: query accuracy vs speed
Index Operator Classes
-- For each distance metric, use matching operator class:
-- L2 distance        → vector_l2_ops
-- Cosine distance    → vector_cosine_ops
-- Inner product      → vector_ip_ops
-- L1 distance        → vector_l1_ops

-- Halfvec (for large dims 2001-4000)
CREATE INDEX ON docs
USING hnsw ((embedding::halfvec(3072)) halfvec_cosine_ops);

-- Binary vectors (bit)
CREATE INDEX ON docs
USING hnsw ((embedding::bit(3072)) bit_hamming_ops);
V5 Similarity Search Queries
Basic KNN Search
-- Find 5 most similar documents
SELECT
  id,
  title,
  1 - (embedding <=> $1)
    AS similarity
FROM documents
ORDER BY embedding <=> $1
LIMIT 5;

-- With minimum similarity threshold
SELECT id, title,
  1 - (embedding <=> $1) AS sim
FROM documents
WHERE 1 - (embedding <=> $1) > 0.8
ORDER BY sim DESC
LIMIT 10;
Filtered Search
-- Pre-filter by metadata then search
SELECT id, title, source
FROM documents
WHERE source = 'manual'
  AND created_at > '2024-01-01'
ORDER BY embedding <=> $1
LIMIT 5;

-- JSONB filter + vector search
SELECT id, title
FROM documents
WHERE meta @> '{"lang":"en"}'
ORDER BY embedding <=> $1
LIMIT 10;
Advanced Patterns
-- Self-similarity (find duplicates)
SELECT a.id, b.id,
  a.embedding <=> b.embedding AS cosine_dist
FROM documents a, documents b
WHERE a.id < b.id
  AND a.embedding <=> b.embedding < 0.05
ORDER BY cosine_dist;

-- Diverse results (avoid clustering)
WITH ranked AS (
  SELECT id, title, category,
    ROW_NUMBER() OVER (
      PARTITION BY category
      ORDER BY embedding <=> $1
    ) AS rn
  FROM documents
)
SELECT * FROM ranked WHERE rn = 1
ORDER BY rn LIMIT 10;
V6 Hybrid Search (Vector + Full-Text)
Hybrid search combines semantic (vector) similarity with keyword (BM25/FTS) relevance for best-of-both results, especially useful in RAG pipelines.
Full-Text Search Setup
-- Add tsvector column
ALTER TABLE documents
ADD COLUMN fts TSVECTOR
  GENERATED ALWAYS AS (
    to_tsvector('english', content)
  ) STORED;

-- GIN index for FTS
CREATE INDEX idx_fts
  ON documents USING GIN(fts);

-- FTS query
SELECT id, title,
  ts_rank(fts, query) AS rank
FROM documents,
  to_tsquery('english','postgres & vector') query
WHERE fts @@ query
ORDER BY rank DESC;
Reciprocal Rank Fusion (RRF)
WITH vec_results AS (
  SELECT id,
    ROW_NUMBER() OVER
      (ORDER BY embedding <=> $1)
    AS vec_rank
  FROM documents
  LIMIT 50
),
fts_results AS (
  SELECT id,
    ROW_NUMBER() OVER
      (ORDER BY ts_rank(fts, $2) DESC)
    AS fts_rank
  FROM documents
  WHERE fts @@ $2
  LIMIT 50
)
SELECT d.id, d.title,
  1.0/(60 + COALESCE(v.vec_rank,1000))
  + 1.0/(60 + COALESCE(f.fts_rank,1000))
  AS rrf_score
FROM documents d
LEFT JOIN vec_results v ON d.id = v.id
LEFT JOIN fts_results f ON d.id = f.id
WHERE v.id IS NOT NULL
  OR f.id IS NOT NULL
ORDER BY rrf_score DESC
LIMIT 10;

Performance Tips
  • Always create an index (HNSW preferred) before running similarity queries on large tables
  • Use HNSW over IVFFlat for better recall; IVFFlat needs data loaded before indexing
  • Normalize vectors before storing if using inner product (<#>) for cosine-like behavior
  • Set ef_search / probes per session based on recall vs latency tradeoff
  • Use LIMIT — KNN is only fast with a LIMIT clause that the planner can push down
  • For >2000 dimensions, use halfvec type to halve memory usage
  • Tune maintenance_work_mem before building HNSW index for speed

Supported Types Summary
vector
Standard 32-bit floats. Up to 16,000 dims. Default type.
halfvec
16-bit floats. Up to 16,000 dims. Half the storage of vector.
bit
Binary vectors. Hamming / Jaccard distance. Very compact.
sparsevec
Sparse representation. Up to 1000 non-zero dims stored.
IVFFlat
Fast build, good for static datasets, tune lists + probes.
HNSW
Best recall, supports live inserts, tune m + ef_search.
