from fastapi import APIRouter, Depends, HTTPException, status from sqlalchemy.ext.asyncio import AsyncSession from app.schemas.user import UserCreate, UserUpdate, UserResponse from app.dependencies import get_db, get_current_user from app.crud import user_crud router = APIRouter() # GET /users ─── list with pagination @router.get("/", response_model=list[UserResponse]) async def list_users(skip: int = 0, limit: int = 100, db: AsyncSession = Depends(get_db)): return await user_crud.get_users(db, skip=skip, limit=limit) # GET /users/{user_id} ─── single resource @router.get("/{user_id}", response_model=UserResponse) async def get_user(user_id: int, db: AsyncSession = Depends(get_db)): user = await user_crud.get_user(db, user_id) if not user: raise HTTPException(status_code=404, detail="User not found") return user # POST /users ─── create (201) @router.post("/", response_model=UserResponse, status_code=status.HTTP_201_CREATED) async def create_user(payload: UserCreate, db: AsyncSession = Depends(get_db)): existing = await user_crud.get_user_by_email(db, payload.email) if existing: raise HTTPException(status_code=409, detail="Email already registered") return await user_crud.create_user(db, payload) # PUT /users/{user_id} ─── full replace (all fields required) @router.put("/{user_id}", response_model=UserResponse) async def replace_user(user_id: int, payload: UserCreate, db: AsyncSession = Depends(get_db)): user = await user_crud.update_user(db, user_id, payload.model_dump()) if not user: raise HTTPException(status_code=404, detail="User not found") return user # PATCH /users/{user_id} ─── partial update (only sent fields) @router.patch("/{user_id}", response_model=UserResponse) async def patch_user(user_id: int, payload: UserUpdate, db: AsyncSession = Depends(get_db)): user = await user_crud.update_user(db, user_id, payload.model_dump(exclude_unset=True)) if not user: raise HTTPException(status_code=404, detail="User not found") return user # DELETE /users/{user_id} ─── 204 No Content @router.delete("/{user_id}", status_code=status.HTTP_204_NO_CONTENT) async def delete_user( user_id: int, db: AsyncSession = Depends(get_db), current_user = Depends(get_current_user), ): deleted = await user_crud.delete_user(db, user_id) if not deleted: raise HTTPException(status_code=404, detail="User not found")
04B

BaseModel Schemas + POST & UPDATE in Depth

This section walks the complete end-to-end process — from defining Pydantic schemas to wiring them into POST, PUT, and PATCH routes with a full SQLAlchemy database layer. Every step is explicit so nothing is hidden.

Step 1 — The SQLAlchemy ORM Model

The ORM model is the source of truth for your database columns, types, constraints, and relationships. It lives in app/models/ and maps 1-to-1 with a database table.

app/models/product.pypython
from sqlalchemy import String, Integer, Float, Boolean, Text, DateTime, ForeignKey, func from sqlalchemy.orm import Mapped, mapped_column, relationship from app.core.database import Base from datetime import datetime class Product(Base): __tablename__ = "products" # Auto-incrementing PK id: Mapped[int] = mapped_column(Integer, primary_key=True, index=True) # Text columns — String(n) = VARCHAR(n), Text = unlimited name: Mapped[str] = mapped_column(String(200), nullable=False) description: Mapped[str | None] = mapped_column(Text, nullable=True) sku: Mapped[str] = mapped_column(String(100), unique=True, index=True) # Numeric price: Mapped[float] = mapped_column(Float, nullable=False) stock: Mapped[int] = mapped_column(Integer, default=0) # Boolean — server_default writes "true" into DB DDL is_active: Mapped[bool] = mapped_column(Boolean, default=True, server_default="true") # Foreign key — links to users.id owner_id: Mapped[int] = mapped_column(ForeignKey("users.id"), nullable=False) # Timestamps — server_default = set by DB on INSERT # onupdate = updated automatically on every UPDATE created_at: Mapped[datetime] = mapped_column(DateTime(timezone=True), server_default=func.now()) updated_at: Mapped[datetime] = mapped_column(DateTime(timezone=True), server_default=func.now(), onupdate=func.now()) # Relationship — lazy="selectin" auto-loads in same query owner: Mapped["User"] = relationship("User", back_populates="products", lazy="selectin")
Column Type SQLAlchemy Python Mapped[] Use When
Short text String(n) Mapped[str] Names, slugs, codes (max length known)
Long text Text Mapped[str] Descriptions, bios, content (no max)
Integer Integer Mapped[int] Counts, IDs, quantities
Decimal money Numeric(10,2) Mapped[Decimal] Prices (avoid Float for money)
Float Float Mapped[float] Scores, ratings, coordinates
Boolean Boolean Mapped[bool] Flags, is_active, is_verified
Timestamp DateTime(timezone=True) Mapped[datetime] created_at, updated_at
Nullable col nullable=True Mapped[str | None] Optional fields
Unique + indexed unique=True, index=True — Email, SKU, slug lookups
Foreign key ForeignKey("table.col") Mapped[int] Linking to another table
ℹ️

Mapped[] vs Column(): Mapped[str] + mapped_column() is the modern SQLAlchemy 2.0 style — it gives full type hints and IDE support. The old Column(String) still works but avoid it in new code.

Step 2 — Define Pydantic Schemas

You need three distinct schemas per resource: one for creation, one for updating, and one for responses. Never reuse the ORM model as a schema — they serve completely different purposes.

app/schemas/product.py — All three schemaspython
from pydantic import BaseModel, ConfigDict, Field, field_validator, model_validator from typing import Optional from datetime import datetime # ──────────────────────────────────────────────────────────── # 1. CREATE — body for POST requests. # Required fields have no default (or Field(...)). # Client must NOT send: id, owner_id, created_at, updated_at # Those are set by the server / DB automatically. # ──────────────────────────────────────────────────────────── class ProductCreate(BaseModel): name: str = Field(..., min_length=1, max_length=200, examples=["Wireless Headphones"]) sku: str = Field(..., pattern=r"^[A-Z0-9\-]+$", examples=["WH-1000XM5"]) price: float = Field(..., gt=0, description="Price in USD, must be > 0") description: Optional[str] = Field(None, max_length=2000) stock: int = Field(0, ge=0) # ge = ≥ 0, default 0 # @field_validator: runs custom Python logic on a field after type coercion @field_validator("name") @classmethod def strip_name(cls, v: str) -> str: v = v.strip() if not v: raise ValueError("Name cannot be blank") return v @field_validator("sku") @classmethod def sku_uppercase(cls, v: str) -> str: return v.upper() # normalise — always store uppercased # ──────────────────────────────────────────────────────────── # 2. UPDATE — body for PUT / PATCH requests. # Every single field is Optional — the client sends ONLY # what they want to change. Use model_dump(exclude_unset=True) # in the handler so un-sent fields are not overwritten. # ──────────────────────────────────────────────────────────── class ProductUpdate(BaseModel): name: Optional[str] = Field(None, min_length=1, max_length=200) price: Optional[float] = Field(None, gt=0) description: Optional[str] = Field(None, max_length=2000) stock: Optional[int] = Field(None, ge=0) is_active: Optional[bool] = None # ──────────────────────────────────────────────────────────── # 3. RESPONSE — what the API returns to the client. # from_attributes=True lets Pydantic read ORM objects directly. # Never expose: hashed_password, internal flags, secret keys. # ──────────────────────────────────────────────────────────── class ProductResponse(BaseModel): model_config = ConfigDict(from_attributes=True) # REQUIRED for ORM → Pydantic id: int name: str sku: str price: float description: Optional[str] stock: int is_active: bool owner_id: int created_at: datetime updated_at: datetime # ── Nested response: product with owner details ──────────── class OwnerBrief(BaseModel): model_config = ConfigDict(from_attributes=True) id: int name: str email: str class ProductWithOwner(ProductResponse): owner: OwnerBrief # nested schema — Pydantic serializes automatically

Field() Constraint Reference

Constraint Syntax Meaning
Required Field(...) or no default Client MUST provide this field
Optional with default Field(None) or = None OK to omit — defaults to None
Min/max string length Field(..., min_length=1, max_length=200) Guards string length
Numeric bounds Field(..., gt=0, ge=0, lt=100, le=100) gt=>, ge=≥, lt=<, le=≤
Regex pattern Field(..., pattern=r"^[A-Z0-9]+") String must match regex
Docs / Swagger Field(..., description="...", examples=[...]) Shows up in /docs UI
Custom validation @field_validator("field_name") Any Python logic
Cross-field validation @model_validator(mode="after") Compare multiple fields
⚠️

Schema vs ORM model — never mix them. ORM models define the database. Pydantic schemas define the API contract. Even if 90% of fields are the same, always define both separately. Mixing them leads to accidentally exposing hashed_password or leaking internal fields.

Here we wire together the database engine, session management, CRUD layer, and route handlers into a complete working stack — step by step.

Database Engine & Session Factory

app/core/database.pypython
from sqlalchemy.ext.asyncio import create_async_engine, AsyncSession, async_sessionmaker from sqlalchemy.orm import DeclarativeBase from app.config import settings # Engine — the connection pool. Create ONCE at startup. # DATABASE_URL must use postgresql+asyncpg:// for async engine = create_async_engine( settings.DATABASE_URL, # e.g. "postgresql+asyncpg://user:pw@host/db" pool_size=10, # connections kept open max_overflow=20, # extra connections under high load pool_pre_ping=True, # verify connection alive before use echo=False, # set True to log all SQL for debugging ) # Session factory — call this to get a new async session AsyncSessionLocal = async_sessionmaker( engine, class_=AsyncSession, expire_on_commit=False, # keep ORM attributes accessible after commit autoflush=False, # we flush manually inside CRUD ) # Base — all ORM models inherit from this class Base(DeclarativeBase): pass # Called at startup via lifespan async def init_db(): async with engine.begin() as conn: await conn.run_sync(Base.metadata.create_all) # In production use Alembic migrations instead of create_all

Session Dependency (get_db)

The get_db dependency opens one session per request, commits on success, and rolls back on any exception. The route handler never touches commit/rollback directly.

app/dependencies.py — get_dbpython
from sqlalchemy.ext.asyncio import AsyncSession from app.core.database import AsyncSessionLocal from typing import AsyncGenerator async def get_db() -> AsyncGenerator[AsyncSession, None]: async with AsyncSessionLocal() as session: try: yield session # handler runs here await session.commit() # success: persist everything except Exception: await session.rollback() # failure: undo all changes raise # re-raise so FastAPI returns 500 # Session closes automatically via async context manager # Usage in any route: # db: AsyncSession = Depends(get_db)

Full Async CRUD Layer

The CRUD module is the only place that writes SQL. Route handlers call CRUD functions — they never call db.execute() directly. This keeps handlers thin and business logic testable.

app/crud/product_crud.py — complete CRUDpython
from sqlalchemy.ext.asyncio import AsyncSession from sqlalchemy import select, update, delete from sqlalchemy.orm import selectinload from app.models.product import Product from app.schemas.product import ProductCreate, ProductUpdate # ────────────────────────────────────────────────────────── # CREATE # model_dump() converts Pydantic schema → plain dict # db.add() stages the INSERT (not committed yet) # flush() sends INSERT to DB + gets auto-generated id back # refresh() re-reads from DB so timestamps/defaults are populated # ────────────────────────────────────────────────────────── async def create_product( db: AsyncSession, payload: ProductCreate, owner_id: int, ) -> Product: product = Product(**payload.model_dump(), owner_id=owner_id) db.add(product) await db.flush() # ← writes to DB within transaction await db.refresh(product) # ← load id, created_at, updated_at return product # ────────────────────────────────────────────────────────── # READ ONE — scalar_one_or_none() returns obj or None # ────────────────────────────────────────────────────────── async def get_product(db: AsyncSession, product_id: int) -> Product | None: result = await db.execute( select(Product).where(Product.id == product_id) ) return result.scalar_one_or_none() # ────────────────────────────────────────────────────────── # READ ONE with nested relationship (owner loaded in 1 query) # ────────────────────────────────────────────────────────── async def get_product_with_owner(db: AsyncSession, product_id: int) -> Product | None: result = await db.execute( select(Product) .options(selectinload(Product.owner)) .where(Product.id == product_id) ) return result.scalar_one_or_none() # ────────────────────────────────────────────────────────── # READ MANY with filters + ordering + pagination # ────────────────────────────────────────────────────────── async def get_products( db: AsyncSession, skip: int = 0, limit: int = 20, active_only: bool = True, owner_id: int | None = None, ) -> list[Product]: stmt = select(Product) if active_only: stmt = stmt.where(Product.is_active == True) if owner_id is not None: stmt = stmt.where(Product.owner_id == owner_id) stmt = stmt.order_by(Product.created_at.desc()).offset(skip).limit(limit) result = await db.execute(stmt) return list(result.scalars().all()) # ────────────────────────────────────────────────────────── # UPDATE — accepts only the fields that were sent # exclude_unset=True means "only fields the client included # in the JSON body". Fields they didn't send are NOT in the # dict and will NOT be overwritten in the DB. # ────────────────────────────────────────────────────────── async def update_product( db: AsyncSession, product_id: int, update_data: dict, # pre-dumped dict from caller ) -> Product | None: if not update_data: return await get_product(db, product_id) # nothing to change await db.execute( update(Product) .where(Product.id == product_id) .values(**update_data) .execution_options(synchronize_session="fetch") ) await db.flush() return await get_product(db, product_id) # ────────────────────────────────────────────────────────── # DELETE — returns True if row existed, False if 404 # ────────────────────────────────────────────────────────── async def delete_product(db: AsyncSession, product_id: int) -> bool: result = await db.execute( delete(Product).where(Product.id == product_id) ) return result.rowcount > 0 # ── Utility: duplicate check ─────────────────────────── async def get_product_by_sku(db: AsyncSession, sku: str) -> Product | None: result = await db.execute(select(Product).where(Product.sku == sku)) return result.scalar_one_or_none()
⚠️

flush() vs commit(): flush() sends SQL to the DB within the open transaction and gives you back auto-generated values (like id), but does NOT make it permanent. commit() makes it permanent. Our get_db() dependency commits on success — so CRUD functions should only call flush(), never commit() directly.

PATCH vs PUT — The Critical Difference

PATCH updates only the fields the client sends. PUT replaces the entire resource. The key is model_dump(exclude_unset=True) — without it, un-sent Optional fields become None and overwrite existing data.

exclude_unset=True — why it matterspython
# ─── Scenario: Client sends ONLY {"price": 249.99} # They want to lower the price. Nothing else should change. ─── payload = ProductUpdate(price=249.99) # only price was set # ❌ WRONG — model_dump() includes ALL fields payload.model_dump() # → {"name": None, "price": 249.99, "description": None, # "stock": None, "is_active": None} # ↑ sending this to DB would NULL OUT name, description, etc! # ✅ CORRECT — exclude_unset skips fields the client didn't send payload.model_dump(exclude_unset=True) # → {"price": 249.99} # ↑ only price is in the dict — name/description stay unchanged in DB # ── exclude_unset vs exclude_none ────────────────────────── # exclude_unset=True → skips fields not included in the JSON at all # exclude_none=True → skips fields whose VALUE is None (even if sent) # # For PATCH: always use exclude_unset=True # For PUT: use model_dump() without exclude — full replacement
app/routers/products.py — PATCH + PUT + DELETE routespython
# ── GET /products/ ─── list all (with filters) ──────────── @router.get("/", response_model=list[ProductResponse]) async def list_products( skip: int = 0, limit: int = 20, active_only: bool = True, db: AsyncSession = Depends(get_db), ): return await product_crud.get_products(db, skip=skip, limit=limit, active_only=active_only) # ── GET /products/{id} ─── single with owner nested ─────── @router.get("/{product_id}", response_model=ProductWithOwner) async def get_product(product_id: int, db: AsyncSession = Depends(get_db)): product = await product_crud.get_product_with_owner(db, product_id) if not product: raise HTTPException(status_code=404, detail="Product not found") return product # ── PATCH /products/{id} ─── partial update ─────────────── @router.patch("/{product_id}", response_model=ProductResponse) async def patch_product( product_id: int, payload: ProductUpdate, # all fields Optional db: AsyncSession = Depends(get_db), current_user: User = Depends(get_current_user), ): product = await product_crud.get_product(db, product_id) if not product: raise HTTPException(status_code=404, detail="Product not found") # Ownership check — only owner or admin may edit if product.owner_id != current_user.id and current_user.role != "admin": raise HTTPException(status_code=403, detail="Access denied") # exclude_unset=True ← THE key for correct PATCH behaviour update_data = payload.model_dump(exclude_unset=True) updated = await product_crud.update_product(db, product_id, update_data) return updated # PUT = replace ALL fields — no exclude_unset update_data = payload.model_dump() return await product_crud.update_product(db, product_id, update_data) # ── DELETE /products/{id} ─── 204 no body ───────────────── @router.delete("/{product_id}", status_code=status.HTTP_204_NO_CONTENT) async def delete_product( product_id: int, db: AsyncSession = Depends(get_db), current_user: User = Depends(get_current_user), ): if current_user.role != "admin": raise HTTPException(status_code=403, detail="Admins only") deleted = await product_crud.delete_product(db, product_id) if not deleted: raise HTTPException(status_code=404, detail="Product not found")

Relationships & Loading Strategies

SQLAlchemy relationships let you traverse across tables. Always specify a loading strategy — the default lazy mode breaks inside async sessions.

Strategy How Queries Use When
selectinload .options(selectinload(Product.owner)) 2 queries (SELECT + IN) Lists — most efficient for collections
joinedload .options(joinedload(Product.owner)) 1 JOIN query Single records with one related object
lazy="selectin" Set on the relationship definition Auto — 2 queries Always needed for async — never use plain lazy
noload .options(noload(Product.owner)) 0 — skips the relation When you explicitly don't need related data
Relationship loading examplespython
from sqlalchemy.orm import selectinload, joinedload, noload # ── Load list of products WITH owner each ───────────────── # Issues: SELECT * FROM products + SELECT * FROM users WHERE id IN (...) # Best for lists where multiple products share owners async def get_products_with_owners(db: AsyncSession): result = await db.execute( select(Product) .options(selectinload(Product.owner)) .where(Product.is_active == True) .order_by(Product.created_at.desc()) .limit(50) ) return result.scalars().all() # ── Load single product WITH owner (JOIN) ───────────────── # Issues: SELECT products JOIN users ON ... WHERE products.id = ? async def get_product_detail(db: AsyncSession, product_id: int): result = await db.execute( select(Product) .options(joinedload(Product.owner)) .where(Product.id == product_id) ) return result.unique().scalar_one_or_none() # .unique() required with joinedload to deduplicate JOIN rows # ── Load user with ALL their products ───────────────────── async def get_user_with_products(db: AsyncSession, user_id: int): result = await db.execute( select(User) .options(selectinload(User.products)) .where(User.id == user_id) ) return result.scalar_one_or_none()

Alembic Migrations

Never use create_all() in production. Use Alembic for versioned, reversible migrations.

Migration workflowbash
# One-time setup uv run alembic init alembic # In alembic/env.py — add these two lines: # from app.core.database import Base # target_metadata = Base.metadata # Generate migration from model changes uv run alembic revision --autogenerate -m "add products table" # Review the generated file in alembic/versions/ before applying! # Apply to DB uv run alembic upgrade head # Rollback one step uv run alembic downgrade -1 # Show current migration state uv run alembic current # Show full history uv run alembic history --verbose

Complete End-to-End Flow Summary

✅

The golden rules: Use ProductCreate (all required) for POST and PUT. Use ProductUpdate (all Optional) for PATCH. Call model_dump(exclude_unset=True) in PATCH — never without it. Always set response_model= to control serialization. Keep DB calls in the CRUD layer, not in route handlers.

Schema Patterns Reference

Middleware

Middleware wraps every request/response. Use it for logging, request IDs, rate limiting, and security headers.

ℹ️

Execution order: Middleware runs in reverse registration order on the way in, and forward on the way out. Register your most important middleware last so it executes first.

app/middleware.pypython
import time, uuid, logging from fastapi import Request, Response, HTTPException from starlette.middleware.base import BaseHTTPMiddleware from collections import defaultdict logger = logging.getLogger("api") # ── Request ID + Timing + Structured Logging ─────────── class RequestLogMiddleware(BaseHTTPMiddleware): async def dispatch(self, request: Request, call_next): request_id = str(uuid.uuid4())[:8] request.state.request_id = request_id start = time.perf_counter() response: Response = await call_next(request) ms = (time.perf_counter() - start) * 1000 response.headers["X-Request-ID"] = request_id response.headers["X-Process-Time-ms"] = f"{ms:.2f}" logger.info(f"{request_id} {request.method} {request.url.path} {response.status_code} {ms:.1f}ms") return response # ── In-memory Rate Limiter ───────────────────────────── _counts: dict = defaultdict(list) class RateLimitMiddleware(BaseHTTPMiddleware): def __init__(self, app, max_req: int = 100, window: int = 60): super().__init__(app) self.max_req = max_req self.window = window async def dispatch(self, request: Request, call_next): ip = request.client.host now = time.time() _counts[ip] = [t for t in _counts[ip] if t > now - self.window] if len(_counts[ip]) >= self.max_req: raise HTTPException(status_code=429, detail="Too Many Requests") _counts[ip].append(now) return await call_next(request) # Register in main.py: # app.add_middleware(RequestLogMiddleware) # app.add_middleware(RateLimitMiddleware, max_req=60, window=60)

JWT Authentication

Access tokens are short-lived (30 min). Refresh tokens are long-lived (30 days) and stored client-side securely. Never store access tokens in localStorage.

app/core/security.pypython
from datetime import datetime, timedelta, timezone from jose import jwt, JWTError from passlib.context import CryptContext from app.config import settings pwd_ctx = CryptContext(schemes=["bcrypt"], deprecated="auto") def hash_password(plain: str) -> str: return pwd_ctx.hash(plain) def verify_password(plain: str, hashed: str) -> bool: return pwd_ctx.verify(plain, hashed) def create_access_token(subject: str, extra: dict = {}) -> str: expire = datetime.now(timezone.utc) + timedelta(minutes=settings.ACCESS_TOKEN_EXPIRE_MINUTES) return jwt.encode({"sub": subject, "exp": expire, "type": "access", **extra}, settings.SECRET_KEY, algorithm="HS256") def create_refresh_token(subject: str) -> str: expire = datetime.now(timezone.utc) + timedelta(days=30) return jwt.encode({"sub": subject, "exp": expire, "type": "refresh"}, settings.SECRET_KEY, algorithm="HS256") def decode_token(token: str) -> dict | None: try: return jwt.decode(token, settings.SECRET_KEY, algorithms=["HS256"]) except JWTError: return None
app/routers/auth.py — Login + Refreshpython
from fastapi import APIRouter, Depends, HTTPException, status from fastapi.security import OAuth2PasswordRequestForm from app.core.security import verify_password, create_access_token, create_refresh_token, decode_token from app.dependencies import get_db from app.crud import user_crud router = APIRouter() @router.post("/token") async def login(form: OAuth2PasswordRequestForm = Depends(), db = Depends(get_db)): user = await user_crud.get_user_by_email(db, form.username) if not user or not verify_password(form.password, user.hashed_password): raise HTTPException(status_code=401, detail="Invalid credentials", headers={"WWW-Authenticate": "Bearer"}) return { "access_token": create_access_token(str(user.id), {"role": user.role}), "refresh_token": create_refresh_token(str(user.id)), "token_type": "bearer", } @router.post("/refresh") async def refresh(refresh_token: str): payload = decode_token(refresh_token) if not payload or payload.get("type") != "refresh": raise HTTPException(status_code=401, detail="Invalid refresh token") return { "access_token": create_access_token(payload["sub"]), "refresh_token": create_refresh_token(payload["sub"]), "token_type": "bearer", }

File & Image Upload

FastAPI handles multipart uploads via UploadFile. Always validate type, size, and sanitize filenames. In production, stream files to S3/R2 — never store on the app server.

File upload with validationpython
import uuid from pathlib import Path from fastapi import APIRouter, UploadFile, File, HTTPException UPLOAD_DIR = Path("uploads") MAX_SIZE = 10 * 1024 * 1024 # 10 MB ALLOWED_TYPES = {"image/jpeg", "image/png", "image/webp", "application/pdf"} @router.post("/upload") async def upload_file(file: UploadFile = File(...)): if file.content_type not in ALLOWED_TYPES: raise HTTPException(400, f"Type {file.content_type!r} not allowed") contents = await file.read() if len(contents) > MAX_SIZE: raise HTTPException(413, "File exceeds 10 MB limit") ext = Path(file.filename).suffix.lower() safe_name = f"{uuid.uuid4()}{ext}" # never trust original filename dest = UPLOAD_DIR / safe_name UPLOAD_DIR.mkdir(exist_ok=True) dest.write_bytes(contents) return {"filename": safe_name, "size": len(contents)} # ── Image resize + convert to WebP ───────────────────── from PIL import Image import io @router.post("/upload/avatar") async def upload_avatar(file: UploadFile = File(...), current_user = Depends(get_current_user)): if not file.content_type.startswith("image/"): raise HTTPException(400, "Images only") contents = await file.read() img = Image.open(io.BytesIO(contents)).convert("RGB") img.thumbnail((256, 256), Image.LANCZOS) buf = io.BytesIO() img.save(buf, format="WEBP", quality=85) # In production: upload buf.getvalue() to S3 here return {"avatar_url": f"/static/avatars/avatar_{current_user.id}.webp"} # ── Multiple files ────────────────────────────────────── @router.post("/upload/bulk") async def upload_bulk(files: list[UploadFile] = File(...)): results = [] for f in files: contents = await f.read() results.append({"name": f.filename, "size": len(contents), "type": f.content_type}) return results

Background Tasks

📬 Built-in BackgroundTasks

from fastapi import WebSocket, WebSocketDisconnect class ConnectionManager: def __init__(self): self.rooms: dict[str, list[WebSocket]] = {} async def connect(self, ws: WebSocket, room: str): await ws.accept() self.rooms.setdefault(room, []).append(ws) def disconnect(self, ws: WebSocket, room: str): self.rooms.get(room, []).remove(ws) async def broadcast(self, room: str, msg: str): for conn in self.rooms.get(room, []): await conn.send_text(msg) manager = ConnectionManager() @app.websocket("/ws/{room}/{username}") async def ws_endpoint(ws: WebSocket, room: str, username: str): await manager.connect(ws, room) await manager.broadcast(room, f"{username} joined") try: while True: data = await ws.receive_text() await manager.broadcast(room, f"{username}: {data}") except WebSocketDisconnect: manager.disconnect(ws, room) await manager.broadcast(room, f"{username} left")
